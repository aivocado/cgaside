def law_work_count(i):
    a = len(df.loc[(df['소관부처명']==i) & (df['사무판단']==0), ])
    b = len(df.loc[(df['소관부처명']==i) & (df['사무판단']==1), ])

    #print(f"소관부처 {i}일때 ========")
    #print(f"사무x: {a}")
    #print(f"사무O: {b}")
    
    result = [b, a, i]
    return result
